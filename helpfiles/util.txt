# Format C code (a la gofmt)
? c,text,edit,lint
% http://astyle.sourceforge.net/

> knr-c-fmt.sh
$ # use astyle to format a C sourcefile in KnR style
$ tmp=$(mktemp)
$ astyle --style=kr --indent=tab < "$1" > "$tmp"
$ mv "$tmp" "$1"
$ rm "$tmp"
--

# Stop and remove ALL docker images on your host
? docker,dangerous,purge,lastresort

> This should only be done in a "well nothing else has worked"
> style situation. It shouldn't break anything but you'll need
> to re-pull and rebuild EVERYTHING...

> purge-docker.sh
$ containers=$(sudo docker ps -aq)
$ images$(sudo docker images -f 'dangling=true' -aq)
$ sudo docker stop $(containers)
$ sudo docker rm $(containers)
$ sudo docker rmi $(images)
--
